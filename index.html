function runMajal(code) { const lines = code.split("\n").map(l => l.trim()).filter(Boolean); let vars = {}; let output = []; let htmlMode = false; let siteName = ""; let html = ""; for (let i = 0; i < lines.length; i++) { let line = lines[i]; // ========== متغيرات ========== if (line.startsWith("خلي")) { let parts = line.replace("خلي", "").split("="); let name = parts[0].trim(); let value = parseInt(parts[1].trim()); vars[name] = value; } // ========== شرط إذا ========== else if (line.startsWith("إذا")) { let condition = line.replace("إذا", "").replace(":", "").trim(); let [left, op, right] = condition.split(" "); let l = vars[left] ?? parseInt(left); let r = vars[right] ?? parseInt(right); let result = false; if (op === "أكبر") result = l > r; if (op === "أصغر") result = l < r; if (op === "يساوي") result = l == r; let nextLine = lines[i + 1]; if (!result) { i++; // تجاهل السطر اللي بعده } } // ========== طباعة ========== else if (line.startsWith("اكتب")) { let text = line.split('"')[1]; output.push(text); } // ========== موقع ========== else if (line.startsWith("موقع")) { siteName = line.split('"')[1]; htmlMode = true; html += `

${siteName}

`; } // ========== منتج ========== else if (line.startsWith("منتج")) { let name = line.split('"')[1]; let price = line.split("سعر")[1].trim(); html += `

${name}

${price} ريال

`; } } return { output, html, vars }; }
